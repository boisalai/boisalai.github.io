    # and return the tuple containing the scores.

    # Appliquer le KMeans
    kmeans = KMeans(n_clusters=k, random_state=0).fit(X)
    y_pred = kmeans.labels_

    # Assigner l'étiquette la plus fréquente à chaque cluster
    labels = numpy.zeros_like(y_pred)
    for i in range(k):
        mask = (y_pred == i)
        labels[mask] = numpy.bincount(y[mask]).argmax()

    # Calculer les métriques
    rand_score = adjusted_rand_score(y, labels)
    mutual_info = adjusted_mutual_info_score(y, labels)
    v_measure = v_measure_score(y, labels)

    return rand_score, mutual_info, v_measure

Information complémentaire​

Trois mesures de performance ont été utilisées :

  1. L'indice de Rand ajusté (Adjusted Rand Index) : mesure la similitude des deux attributions, en ignorant les permutations et en tenant compte des hasards.
  2. Le score basé sur l'information mutuelle (Adjusted Mutual Information) : mesure l'information partagée entre les étiquettes vraies et celles prédites, en ajustant le hasard.
  3. La mesure V (V Measure) : évalue l'homogénéité et la complétude, où l'homogénéité signifie que chaque cluster contient des membres d'une seule classe, et la complétude signifie que tous les membres d'une classe donnée sont attribués au même cluster.

    # TODO Q1B
    # Mesurez les différentes métriques avec les fonctions
    # Scikit-learn et retournez le tuple qui contient
    # les scores.
    # Measure the different metrics using the functions
    # Scikit-learn and return the tuple containing the scores.

    # Ajuster le modèle de mélange gaussien
    gmm = GaussianMixture(n_components=k, init_params=init, random_state=0).fit(X)

    # Prédir les probabilités pour chaque composant
    probs = gmm.predict_proba(X)

    # Calculer de P(C_i|G_j)
    P_Ci_Gj = numpy.zeros((2, k))
    for j in range(k):
        for i in range(2):
            P_Ci_Gj[i, j] = numpy.sum(probs[y == i, j]) / numpy.sum(probs[:, j])

    # Calculer les probabilités à posteriori P(C_i|x^t)
    P_Ci_xt = numpy.dot(P_Ci_Gj, probs.T) / numpy.dot(P_Ci_Gj, probs.T).sum(axis=0)

    # Attribuer à la classe avec la plus haute probabilité à posteriori
    y_pred = numpy.argmax(P_Ci_xt, axis=0)

    # Calculer les métriques
    rand_score = adjusted_rand_score(y, y_pred)
    mutual_info = adjusted_mutual_info_score(y, y_pred)
    v_measure = v_measure_score(y, y_pred)

    return rand_score, mutual_info, v_measure

    ratios = []

        intra_class_distances = cdist(X[y == class_label], X[y == class_label])

        ratios.append((class_label, ratio))
