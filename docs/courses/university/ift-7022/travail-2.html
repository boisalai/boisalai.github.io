Travail pratique 2

Échéance : À remettre le 24 nov. 2023 à 23h59. Je suis l'équipe 43.

Nous avons trois fichiers contenant des descriptions d'accidents.

  • un fichier d’entraînement : data/incidents_train.json
  • un fichier de validation : data/incidents_dev.json
  • un fichier de test : data/incidents_test.json

Chaque fichier contient deux colonnes :

  • colonne text qui est une chaîne de caractère correspondant à la description d'un accident.
  • colonne label qui est un entier de 0 à 8 correspondant à l'étiquette du type d'accident.

Nous devons entraîner un modèle de réseau de neurones de type feedforward multicouche (MLP) avec plongements de mots pour déterminer le type d’un incident à partir de sa description.

Nous devons respecter les consignes suivantes :

  • Utilisation de Spacy pour la tokenisation et les plongements de mots.
  • Aucune normalisation des données.
  • Agrégation des plongements de mots : Comparer les approches max, average et min pooling.
  • Structure du réseau : une seule couche cachée mais nous devons choisir la taille.
  • Explication de la démarche utilisée.
  • Le code doit être structuré selon les étapes suivantes :
      1. Création du jeu de données (dataset)
      1. Gestion de plongements de mots (embeddings)
      1. Création de modèle(s)
      1. Fonctions utilitaires
      1. Entraînement de modèle(s)
      1. Évaluation et analyse de résultats

import json

# Chargement des fichiers
def load_data(file_path):
    with open(file_path, 'r', encoding='utf-8') as file:
        data_list = json.load(file)

    # Utilisation de la compréhension de liste pour extraire les textes et les labels
    texts = [item['text'] for item in data_list]
    labels = [int(item['label']) for item in data_list]

    return texts, labels

# Étape 2. Gestion de plongements de mots (embeddings)¶

import spacy 

nlp = spacy.load('en_core_web_md')

import numpy as np

def average_embedding(sentence, nlp_model=nlp):
    tokenised_sentence = nlp_model(sentence)  # tokenized
    nb_column = len(tokenised_sentence)
    nb_rows =  nlp_model.vocab.vectors_length
    sentence_embedding_matrix = np.zeros((nb_rows, nb_column))                                  
    for index, token in enumerate(tokenised_sentence):
        sentence_embedding_matrix[:, index] = token.vector
    return np.average(sentence_embedding_matrix, axis=1)

def maxpool_embedding(sentence, nlp_model=nlp): 
    tokenised_sentence = nlp_model(sentence)
    nb_column = len(tokenised_sentence)
    nb_rows =  nlp_model.vocab.vectors_length 
    sentence_embedding_matrix = np.zeros((nb_rows, nb_column))                                    
    for index, token in enumerate(tokenised_sentence):
        sentence_embedding_matrix[:, index] = token.vector
    return np.max(sentence_embedding_matrix, axis=1)

def minpool_embedding(sentence, nlp_model=nlp): 
    tokenised_sentence = nlp_model(sentence)
    nb_column = len(tokenised_sentence)
    nb_rows =  nlp_model.vocab.vectors_length 
    sentence_embedding_matrix = np.zeros((nb_rows, nb_column))                                    
    for index, token in enumerate(tokenised_sentence):
        sentence_embedding_matrix[:, index] = token.vector
    return np.min(sentence_embedding_matrix, axis=1)

def spacy_embedding(sentence, nlp_model=nlp):
    tokenised_sentence = nlp_model(sentence)
    return tokenised_sentence.vector

# Étape 3. Création de modèle(s)

import torch
import torch.nn as nn
import torch.nn.functional as F

class MultiLayerPerceptron(nn.Module):
    def __init__(self, input_size, hidden_layer_size, output_size) :
        super(MultiLayerPerceptron, self).__init__()
        self.fc1 = nn.Linear(input_size, hidden_layer_size)
        self.fc2 = nn.Linear(hidden_layer_size, output_size)
        
    def forward(self, x):
        x = F.relu(self.fc1(x))
        x = self.fc2(x)
        return x

from torch.utils.data import Dataset, DataLoader
from torch import FloatTensor, LongTensor
from typing import List

class SpacyDataset(Dataset):
    def __init__(self, dataset: List[str] , target: np.array, sentence_aggregation_function):
        self.dataset = dataset
        self.doc_embeddings = [None for _ in range(len(dataset))]
        self.sentence_aggregation_function = sentence_aggregation_function 
        self.target = target
    
    def __len__(self):
        return len(self.dataset)

    def __getitem__(self, index):
        if self.doc_embeddings[index] is None:
            self.doc_embeddings[index] = self.sentence_aggregation_function(self.dataset[index])
        return FloatTensor(self.doc_embeddings[index]), LongTensor([self.target[index]]).squeeze(0)

# Fonction pour la phase d'entraînement
def train_model(model, criterion, optimizer, train_loader, valid_loader, num_epochs):
    
    # Pour stocker les pertes et les précisions pour chaque époque
    history = {
        "train_loss": [],
        "valid_loss": [],
        "train_acc": [],
        "valid_acc": []
    }
    
    for epoch in range(num_epochs):
        model.train()  # Définissez le modèle en mode d'entraînement

        # Suivi des mesures
        running_loss = 0.0
        correct_predictions = 0
        total_predictions = 0

        for batch_idx, (inputs, targets) in enumerate(train_loader):
            optimizer.zero_grad()  # Effacer les gradients précédents
            inputs, targets = inputs.to(device), targets.to(device)

            # Avance directe
            outputs = model(inputs)
            loss = criterion(outputs, targets)

            # Rétropropagation et optimisation
            optimizer.zero_grad()
            loss.backward()
            optimizer.step()

            # Mettre à jour les statistiques
            running_loss += loss.item()
            _, predicted = outputs.max(1)
            correct_predictions += predicted.eq(targets).sum().item()
            total_predictions += targets.size(0)

        # Appeler le scheduler à la fin de chaque époque
        scheduler.step()

        # Après la boucle d'entraînement, calculer la perte et la précision moyennes
        epoch_loss = running_loss / len(train_loader)
        epoch_acc = (correct_predictions / total_predictions) * 100.0

        # Ajouter la perte et la précision à l'historique
        history['train_loss'].append(epoch_loss)
        history['train_acc'].append(epoch_acc)

        # Valider le modèle et obtenir la perte et la précision
        valid_loss, valid_acc = validate_model(model, criterion, valid_loader)

        # Ajouter la perte et la précision de validation à l'historique
        history['valid_loss'].append(valid_loss)
        history['valid_acc'].append(valid_acc)

        print(f"Epoch {epoch+1}/{num_epochs}, " 
                + f"Loss: {epoch_loss:.4f}, Accuracy: {epoch_acc:.2f}%, " 
                + f"Val_Loss: {valid_loss:.4f}, Val_Accuracy: {valid_acc:.2f}%")

    return history


# Fonction pour la phase de validation
def validate_model(model, criterion, valid_loader):
    model.eval()  # Définissez le modèle en mode d'évaluation

    valid_loss = 0.0
    correct_predictions = 0
    total_predictions = 0

    with torch.no_grad():  # Désactivez la gradation pour l'évaluation, économisez de la mémoire et des calculs
        for inputs, targets in valid_loader:
            inputs, targets = inputs.to(device), targets.to(device)

            # Avance directe
            outputs = model(inputs)
            loss = criterion(outputs, targets)

            # Mettre à jour les statistiques
            valid_loss += loss.item()
            _, predicted = outputs.max(1)
            correct_predictions += predicted.eq(targets).sum().item()
            total_predictions += targets.size(0)

    # Calculer la perte et l'exactitude.
    epoch_loss = valid_loss / len(valid_loader)
    epoch_acc = (correct_predictions / total_predictions) * 100.0
    return epoch_loss, epoch_acc
