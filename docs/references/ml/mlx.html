  • MLX repo

Install MLX​

python3 -m venv .env
source .env/bin/activate  # Linux or macOS
# .env\Scripts\activate  # Windows
python3 -m pip install --upgrade pip
pip install mlx

See Build and Install MLX.

In Visual Studio Code, open the project, select "Command Palette" from the "View" menu (Cmd+Shift+P on Mac or Ctrl+Shift+P on Windows), run the "Python: Select Interpreter" command, and select the Python interpreter in the virtual directory (.env/bin/python3.9).

Create a main.py file and this code.

>> import mlx.core as mx
>> a = mx.array([1, 2, 3, 4])
>> print(a.shape)
[4]

You should see this in your terminal in Visual Studio Code.

Matrix multiplication​

import time
import mlx.core as mx

start_time = time.time()

def fun(a, b, d1, d2):
  x = mx.matmul(a, b, stream=d1)
  for _ in range(500):
      b = mx.exp(b, stream=d2)
  return x, b

a = mx.random.uniform(shape=(4096, 512))
b = mx.random.uniform(shape=(512, 4))
d1 = mx.gpu
d2 = mx.cpu
x, b = fun(a, b, d1, d2)

end_time = time.time()
elapsed_time = (end_time - start_time) * 1000
print(f"Elapsed time: {elapsed_time} ms")

Gradient checkpoint transform​

2024-02-01

@mx.checkpoint
def fun(x):
    return mx.sum(mx.exp(x))

# Needs less memory!
x = mx.random.uniform(shape=(4096, 512))
dfdx = mx.grad(fun)(x)

from safetensors.mlx import save_file

The previous code should generate this output.

==========
Prompt: #write a quick sort algorithm


def quick_sort(arr):
    if len(arr) <= 1:
        return arr
    pivot = arr[0]
    left = []
    right = []
    for i in range(1, len(arr)):
        if arr[i] < pivot:
            left.append(arr[i])
        else:
            right.append(arr[i])
    return quick_sort(left)

Prompt: #write a quick sort algorithm in VBA\Excel


Sub QuickSort(arr As Variant, l As Long, r As Long)
    Dim pivot As Long
    If l < r Then
        pivot = Partition(arr, l, r)
        QuickSort arr, l, pivot - 1
        QuickSort arr, pivot + 1, r
    End If
End Sub
