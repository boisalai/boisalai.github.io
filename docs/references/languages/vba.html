Option Explicit
 
'credit for this technique goes to John Walkenback
'http://j-walk.com/ss/excel/tips/tip82.htm
Sub GetDataDemo()
 
      Dim FilePath$, Row&, Column&, Address$
     
      'change constants & FilePath below to suit
      '***************************************
      Const FileName$ = "Book1.xls"
      Const SheetName$ = "Sheet1"
      Const NumRows& = 10
      Const NumColumns& = 10
      FilePath = ActiveWorkbook.Path & "\"
      '***************************************
     
      DoEvents
      Application.ScreenUpdating = False
      If Dir(FilePath & FileName) = Empty Then
            MsgBox "The file " & FileName & " was not found", , "File Doesn't Exist"
            Exit Sub
      End If
      For Row = 1 To NumRows
            For Column = 1 To NumColumns
                  Address = Cells(Row, Column).Address
                  Cells(Row, Column) = GetData(FilePath, FileName, SheetName, Address)
                  Columns.AutoFit
            Next Column
      Next Row
      ActiveWindow.DisplayZeros = False
End Sub
 
Private Function GetData(Path, File, Sheet, Address)
      Dim Data$
      Data = "'" & Path & "[" & File & "]" & Sheet & "'!" & _
            Range(Address).Range("A1").Address(, , xlR1C1)
      GetData = ExecuteExcel4Macro(Data)
End Function

Sub BoutonBrouillonsClic()
    Dim ol As Outlook.Application
    Dim mail As Outlook.MailItem

    Dim shtMessage As Worksheet

    Dim signature As String

    Set shtMessage = Workbooks(1).Worksheets("Message")
    
    rowIndex = 2
    Feuil1.prgBar.Max = sht.Cells.Find("*", [A1], , , xlByRows, xlPrevious).Row - 1
    Feuil1.prgBar.Value = 0

    Set ol = New Outlook.Application
    
    Set mail = ol.CreateItem(olMailItem)
    mail.Display
    'signature = mail.HTMLBody

        If Dir$(Range("CheminDestination").Value & "\" & sht.Cells(rowIndex, 7).Value) <> "" Then

            mail.CC = sht.Cells(rowIndex, 4).Value
            mail.Subject = shtMessage.Cells(4, 2).Value
            mail.HTMLBody = "<div style=""font: 1em Arial;"">" & Replace(Replace(Replace(Replace(shtMessage.Cells(6, 2).Value, "[NoProjet]", sht.Cells(rowIndex, 2)), "[Date fin]", Range("DateFinPeriode").Value), "[NomProjet]", sht.Cells(rowIndex, 3)), vbLf, "<br>") & "</div>" & signature
            ' "<div style=""font: 1em Arial;"">" & shtMessage.Cells(6, 2).Value & "</div>" & signature
            mail.SentOnBehalfOfName = "infrastructures@education.gouv.qc.ca"
            mail.Save
            mail.Attachments.Add Range("CheminDestination").Value & "\" & sht.Cells(rowIndex, 5).Value
            mail.Attachments.Add Range("CheminDestination").Value & "\" & sht.Cells(rowIndex, 6).Value
            mail.Attachments.Add Range("CheminDestination").Value & "\" & sht.Cells(rowIndex, 7).Value
            mail.Close olSave
        End If
        rowIndex = rowIndex + 1

Sub GetDataFromADO()  
'Declare variables'   
Set objMyConn = New ADODB.Connection    
Set objMyRecordset = New ADODB.Recordset    
Dim strSQL As String

Obtenir l'espace mémoire RAM​

Function SysMemory()
    Dim oInstance
    Dim colInstances
    Dim dRam As Double
    Set colInstances = GetObject("winmgmts:").ExecQuery("SELECT * FROM Win32_PhysicalMemory")
    For Each oInstance In colInstances
        dRam = dRam + oInstance.Capacity
    Next
    SysMemory = dRam / 1024 / 1024 & "MB"
End Function

Sub ListFiles()
    Dim fs As Object
    Dim folder As Object
    Dim subfolder As Object
    Dim file As Object
    Dim Row As Integer

    Set fs = CreateObject("Scripting.FileSystemObject")

    ' Changez ceci par le chemin de votre répertoire racine
    Set folder = fs.GetFolder(\\chemin\vers\le\répertoire\racine)

    ' Commencer à écrire à partir de la deuxième ligne
    Row = 2

    ' En-têtes pour les données
    With ThisWorkbook.Sheets(1)
        .Cells(1, 1).Value = "Chemin du répertoire"
        .Cells(1, 2).Value = "Dernière Date de Mise à Jour"
        .Cells(1, 3).Value = "Nombre de Fichiers"
        .Cells(1, 4).Value = "Dimension Totale (Mo)"
    End With

    ' Appeler la fonction récursive pour parcourir les répertoires
    Call ListFolderDetails(folder, Row)
    MsgBox "Liste complétée!"
End Sub

' Fonction récursive pour parcourir les répertoires
Sub ListFolderDetails(folder As Object, ByRef Row As Integer)
    Dim subfolder As Object
    Dim file As Object
    Dim fileSize As Double
    Dim latestDate As Date
    Dim fileCount As Long
    fileSize = 0
    latestDate = DateSerial(1900, 1, 1)

    ' Compter les fichiers et accumuler la taille
    For Each file In folder.Files
        fileSize = fileSize + file.Size
        fileCount = fileCount + 1
        If file.DateLastModified > latestDate Then
            latestDate = file.DateLastModified
        End If
    Next file

    ' Convertir la taille en Mo
    fileSize = fileSize / 1024 / 1024

    ' Écrire les détails dans Excel
    With ThisWorkbook.Sheets(1)
        .Cells(Row, 1).Value = folder.Path
        .Cells(Row, 2).Value = latestDate
        .Cells(Row, 3).Value = fileCount
        .Cells(Row, 4).Value = fileSize
    End With

    ' Incrémenter la ligne
    Row = Row + 1

    ' Parcourir les sous-dossiers
    For Each subfolder In folder.SubFolders
        Call ListFolderDetails(subfolder, Row)
    Next subfolder
End Sub
